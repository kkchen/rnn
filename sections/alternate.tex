    \begin{itemize}
        \item<+-> Physics/engineering analogy: time analogous to 1-D space
        \item Ordinary differential equation methods translate between the two
    \end{itemize}
    \uncover<+->{Can use 1-D CNN{}s to process sequences}

    \begin{center}
